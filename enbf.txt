program : statements

statements : statement
           | statements statement
           ;
           
statement : attribution PONTO
          | expression PONTO
          | ifelse PONTO
          | declaration PONTO
          | while PONTO
          | print PONTO
          ;
 
attribution : NOME IG expression ;

expression : term
           | term MAIS term
           | term MENOS term
           | term OU term
           ;

ifelse : SE PABR relexpression PFEC INICIO statements FIM
       | SE PABR relexpression PFEC INICIO statements FIM SENAO INICIO statements FIM
       ;

declaration : type NOME

while : ENQUANTO relexpression FAZER statement ENCERRADO ;

print: MOSTRAR PABR expression PFEC

relexpression : expression IGUAL expression
              | expression DIFER expression
              | expression MAIOR expression
              | expression MENOR expression
              | expression MAIORIG expression
              | expression MENORIG expression
              
term : factor
     | factor MULTIP factor
     | factor DIVIDE factor
     | factor ETMB factor
     ;
    
factor : NUMERO
       | NOME
       | PABR expression PFEC
       ;
     
type : INTEGER ;

dummy :  ;
